from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from datetime import datetime
import uvicorn, json, os

# =========================
# âš™ï¸ CONFIGURACIÃ“N INICIAL
# =========================
DATA_FILE = "data.json"

app = FastAPI(
    title="RubÃ©nWare API Template v2",
    description="Plantilla base de FastAPI con persistencia en JSON",
    version="2.0.0"
)

# =========================
# ðŸ“¦ MODELOS DE DATOS
# =========================
class Item(BaseModel):
    id: int | None = None
    name: str
    description: str | None = None
    price: float
    tax: float | None = None

# =========================
# ðŸ’¾ FUNCIONES DE PERSISTENCIA
# =========================
def load_data() -> list[dict]:
    if not os.path.exists(DATA_FILE):
        return []
    with open(DATA_FILE, "r", encoding="utf-8") as f:
        try:
            return json.load(f)
        except json.JSONDecodeError:
            return []

def save_data(data: list[dict]):
    with open(DATA_FILE, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=4)

# =========================
# ðŸ“š ENDPOINTS DE SISTEMA
# =========================
@app.get("/")
async def root():
    return {"message": "API RubÃ©nWare v2 funcionando ðŸ”¥", "timestamp": datetime.now().isoformat()}

@app.get("/info")
async def info():
    return {
        "status": "online",
        "framework": "FastAPI",
        "version": "2.0.0",
        "data_file": os.path.abspath(DATA_FILE),
        "timestamp": datetime.now().isoformat()
    }

# =========================
# ðŸ§© CRUD PERSISTENTE
# =========================
@app.get("/items/", response_model=list[Item])
async def read_items():
    return load_data()

@app.get("/items/{item_id}", response_model=Item)
async def read_item(item_id: int):
    data = load_data()
    for item in data:
        if item["id"] == item_id:
            return item
    raise HTTPException(status_code=404, detail="Item no encontrado")

@app.post("/items/", response_model=Item)
async def create_item(item: Item):
    data = load_data()
    item.id = (max([i["id"] for i in data]) + 1) if data else 1
    data.append(item.dict())
    save_data(data)
    return item

@app.put("/items/{item_id}", response_model=Item)
async def update_item(item_id: int, updated_item: Item):
    data = load_data()
    for index, item in enumerate(data):
        if item["id"] == item_id:
            updated_item.id = item_id
            data[index] = updated_item.dict()
            save_data(data)
            return updated_item
    raise HTTPException(status_code=404, detail="Item no encontrado")

@app.delete("/items/{item_id}")
async def delete_item(item_id: int):
    data = load_data()
    for index, item in enumerate(data):
        if item["id"] == item_id:
            del data[index]
            save_data(data)
            return {"message": f"Item {item_id} eliminado correctamente"}
    raise HTTPException(status_code=404, detail="Item no encontrado")

# =========================
# ðŸš€ EJECUCIÃ“N LOCAL
# =========================
if __name__ == "__main__":
    uvicorn.run("main:app", host="127.0.0.1", port=8000, reload=True)
